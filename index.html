ditosis.config.py 
from ditosis import DataGenerator, Config

config = Config(
    data_type="text",
    format="conversation",
    samples=1_000_000,
    languages=["en", "es", "zh"],
    distribution={
        "casual": 0.4,
        "technical": 0.3,
        "formal": 0.3
    }
)

generator = DataGenerator(config)
dataset = generator.generate()

# Quality validation
report = dataset.validate()
print(f"Quality Score: {report.score}%")

dataset.export("s3://your-bucket/training-data/")
